    def CohentEffectSize(group1,group2):
      diff = group1.mean()-group2.mean()
      var1 = group1.var()
      var2 = group2.var()
      n1,n2 = len(group1),len(group2)
      pooled_var = (n1*var1+n2*var2)/(n1+n2)
      pooled_std = math.sqrt(pooled_var)
      d = diff/pooled_std
